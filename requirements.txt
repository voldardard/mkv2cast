# mkv2cast Python dependencies
# =============================
#
# These packages are OPTIONAL but recommended for the best experience.
# mkv2cast will work without them but with reduced functionality.
#
# Install with: pip install -r requirements.txt

# Rich terminal UI - provides beautiful progress bars and tables
# Without this, mkv2cast falls back to a basic text-based progress display
rich>=13.0.0

# TOML configuration file support (for Python < 3.11)
# Python 3.11+ has built-in tomllib and doesn't need this
# Without this, mkv2cast uses INI format configuration instead
tomli>=2.0.0; python_version < "3.11"
